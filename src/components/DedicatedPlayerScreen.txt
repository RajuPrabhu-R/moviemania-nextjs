import React, { useEffect, useState, useCallback } from "react";
import { Play, ArrowLeft, ChevronDown, SkipForward } from "lucide-react";
// ⬅️ CRITICAL IMPORTS
import { useNavigate, useSearchParams } from "react-router-dom";

const API_KEY = import.meta.env.VITE_TMDB_API_KEY;

// -------------------------------------------------------------------
// HELPER: Naruto Shippuden Episode Range Map
// -------------------------------------------------------------------
const NARUTO_SHIPPUDEN_START_EPISODES = {
    1: 1, 2: 33, 3: 54, 4: 72, 5: 89, 6: 113, 
    7: 144, 8: 152, 10: 197, 12: 243, 13: 276, 
    14: 296, 15: 321,
};
const NARUTO_SHIPPUDEN_END_EPISODES = {
    1: 32, 2: 53, 3: 71, 4: 88, 5: 112, 6: 143, 
    7: 151, 8: 175, 10: 221, 12: 275, 13: 295,
    14: 320, 15: 348
};

// 1. EPISODES LIST COMPONENT (Nested Helper) 🌟 CLEANUP
// -------------------------------------------------------------------
// Removed unused 'isAnilist' prop. Passed 'onEpisodeClick' and 'onSeasonSelect' as they are required.
const EpisodesList = React.memo(({ tvId, seasons, onEpisodeClick, onSeasonSelect, currentPlayingSeasonNumber }) => {
    if (!tvId) return null;

    const [selectedSeason, setSelectedSeason] = useState(null);
    const [episodes, setEpisodes] = useState([]);
    const [loadingEpisodes, setLoadingEpisodes] = useState(false);
    const [error, setError] = useState(null);
    const [isSeasonListOpen, setIsSeasonListOpen] = useState(false); 

    const MAX_SEASON_NAME_LENGTH = 20;

    const fetchEpisodes = useCallback(async (seasonNumber) => {
        if (!tvId || !seasonNumber) return;

        setLoadingEpisodes(true);
        setError(null);
        try {
            const res = await fetch(
                `${API_BASE_URL}/tv/${tvId}/season/${seasonNumber}?api_key=${API_KEY}`
            );
            const data = await res.json();

            if (data.success === false) {
                setError("Failed to load episodes for this season.");
                setEpisodes([]);
                const currentSeasonData = seasons.find(s => s.season_number === seasonNumber) || null;
                onSeasonSelect(currentSeasonData);
                return;
            }

            setEpisodes(data.episodes || []);
            // Pass the entire season details object back to the parent
            onSeasonSelect(data); 

        } catch (err) {
            setError("Error fetching episodes.");
        } finally {
            setLoadingEpisodes(false);
        }
    }, [tvId, seasons, onSeasonSelect]);

    // Effect 1: Fetch episodes whenever selectedSeason changes
    useEffect(() => {
        if (selectedSeason) {
            fetchEpisodes(selectedSeason.season_number);
        }
    }, [selectedSeason, fetchEpisodes]);

    // Effect 2: Set initial season on mount or when seasons change
    useEffect(() => {
        if (seasons && seasons.length > 0) {
            // Priority: Use the season number from the URL/parent if available, otherwise default to first season > 0
            const initialSeasonNumber = currentPlayingSeasonNumber || 
                                        seasons.find(s => s.season_number > 0)?.season_number || 
                                        seasons[0].season_number;

            const initialSeason = seasons.find(s => s.season_number === initialSeasonNumber);

            // Only update if a valid initial season is found and it's different from the current
            if (initialSeason && initialSeason.id !== selectedSeason?.id) {
                setSelectedSeason(initialSeason);
            }
        }
    // No dependency on selectedSeason here to avoid infinite loop with Effect 3
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [seasons]); 

    // 🌟 Effect 3: SYNCHRONIZATION FIX (Updates selectedSeason when parent state changes via URL)
    useEffect(() => {
        if (currentPlayingSeasonNumber && selectedSeason?.season_number !== currentPlayingSeasonNumber) {
            const seasonToSelect = seasons.find(s => s.season_number === currentPlayingSeasonNumber);
            if (seasonToSelect) {
                setSelectedSeason(seasonToSelect);
            }
        }
    }, [currentPlayingSeasonNumber, seasons, selectedSeason]);


    const availableSeasons = seasons.filter(s => s.season_number !== 0);

    if (!availableSeasons.length) {
        return <div className="text-gray-400">No season information available.</div>;
    }

    const formatSeasonName = (name) => {
        if (!name) return '';
        if (name.length > MAX_SEASON_NAME_LENGTH) {
            return name.slice(0, MAX_SEASON_NAME_LENGTH - 3) + '...';
        }
        return name;
    };

    const handleSeasonChange = (season) => {
        setSelectedSeason(season);
        setIsSeasonListOpen(false); // Close the list after selection
    };

    // 🌟 Enhancement: Memoize the episode click handler for better performance
    const handleEpisodeClick = useCallback((episode) => {
        onEpisodeClick(episode);
    }, [onEpisodeClick]);

    return (
        <div>
            {/* CUSTOM STYLED SEASON SELECTOR */}
            <div className="relative inline-block text-left mb-6">
                <button
                    className="flex items-center justify-between w-full bg-white/10 text-white py-2 pl-4 pr-3 rounded-lg hover:bg-white/20 transition-colors min-w-48"
                    onClick={() => setIsSeasonListOpen(prev => !prev)}
                >
                    <span className="truncate">{selectedSeason ? formatSeasonName(selectedSeason.name) : 'Select Season'}</span>
                    <ChevronDown className={`w-4 h-4 ml-2 transition-transform ${isSeasonListOpen ? 'rotate-180' : 'rotate-0'}`} />
                </button>

                {isSeasonListOpen && (
                    <div className="absolute left-0 mt-1 w-full max-h-60 overflow-y-auto bg-gray-900 border border-pink-600 rounded-lg shadow-lg z-20">
                        {availableSeasons.map((season) => (
                            <div
                                key={season.id}
                                className={`py-2 px-4 cursor-pointer transition-colors ${
                                    selectedSeason && selectedSeason.id === season.id
                                        ? 'bg-pink-600 font-semibold' 
                                        : 'hover:bg-white/10' 
                                }`}
                                onClick={() => handleSeasonChange(season)}
                                title={season.name}
                            >
                                <span className="truncate">
                                    {formatSeasonName(season.name)}
                                </span>
                            </div>
                        ))}
                    </div>
                )}
            </div>

            {loadingEpisodes && <div className="text-pink-500">Loading episodes...</div>}
            {error && <div className="text-red-500">{error}</div>}

            {!loadingEpisodes && !error && (
                <div className="space-y-4">
                    {episodes.map((episode) => (
                        <div 
                            key={episode.id} 
                            className="flex gap-4 p-3  rounded-lg border border-gray-800 hover:bg-white/10 transition cursor-pointer"
                            // Use memoized handler
                            onClick={() => handleEpisodeClick(episode)} 
                        >
                            <div className="relative flex-shrink-0 w-32 h-20 rounded-md overflow-hidden bg-black">
                                <img
                                    src={
                                        episode.still_path
                                            ? `https://image.tmdb.org/t/p/w300${episode.still_path}`
                                            : '/No-Poster.png'
                                    }
                                    alt={`S${episode.season_number} E${episode.episode_number}`}
                                    className="w-full h-full object-cover"
                                />
                                <div className="absolute inset-0 flex items-center justify-center bg-black/40 hover:bg-black/60 transition">
                                    <Play size={24} className="text-white fill-current" />
                                </div>
                            </div>

                            <div className="flex-grow">
                                <h3 className="font-semibold text-xs md:text-xs lg:text-base">
                                    {episode.episode_number}. {episode.name}
                                </h3>
                                <p className="text-xs hidden sm:hidden md:hidden lg:block text-gray-400 mt-1 line-clamp-2">
                                    {episode.overview || 'No description available.'}
                                </p>
                            </div>

                            <div className="flex-shrink-0 text-right">
                                {/* 🌟 Fixed: Added check for episode.runtime to prevent NaN/null errors */}
                                {episode.runtime !== null && episode.runtime !== undefined && (
                                    <p className="text-xs text-gray-400 md:text-xs">
                                        {Math.floor(episode.runtime / 60)}h {episode.runtime % 60}m
                                    </p>
                                )}
                            </div>
                        </div>
                    ))}
                </div>
            )}

             {!loadingEpisodes && !error && episodes.length === 0 && (
                <div className="text-gray-400">No episodes found for this season.</div>
            )}
        </div>
    );
});

const EpisodesList = React.memo(({ tvId, seasons, onEpisodeClick, onSeasonSelect, currentPlayingSeasonNumber }) => {
    if (!tvId) return null;

    const [selectedSeason, setSelectedSeason] = useState(null);
    const [episodes, setEpisodes] = useState([]);
    const [loadingEpisodes, setLoadingEpisodes] = useState(false);
    const [error, setError] = useState(null);
    const [isSeasonListOpen, setIsSeasonListOpen] = useState(false); 

    const MAX_SEASON_NAME_LENGTH = 20;

    const fetchEpisodes = useCallback(async (seasonNumber) => {
        if (!tvId || !seasonNumber) return;

        setLoadingEpisodes(true);
        setError(null);
        try {
            const res = await fetch(
                `${API_BASE_URL}/tv/${tvId}/season/${seasonNumber}?api_key=${API_KEY}`
            );
            const data = await res.json();

            if (data.success === false) {
                setError("Failed to load episodes for this season.");
                setEpisodes([]);
                const currentSeasonData = seasons.find(s => s.season_number === seasonNumber) || null;
                onSeasonSelect(currentSeasonData);
                return;
            }

            setEpisodes(data.episodes || []);
            // Pass the entire season details object back to the parent
            onSeasonSelect(data); 

        } catch (err) {
            setError("Error fetching episodes.");
        } finally {
            setLoadingEpisodes(false);
        }
    }, [tvId, seasons, onSeasonSelect]);

    // Effect 1: Fetch episodes whenever selectedSeason changes
    useEffect(() => {
        if (selectedSeason) {
            fetchEpisodes(selectedSeason.season_number);
        }
    }, [selectedSeason, fetchEpisodes]);

    // Effect 2: Set initial season on mount or when seasons change
    useEffect(() => {
        if (seasons && seasons.length > 0) {
            // Priority: Use the season number from the URL/parent if available, otherwise default to first season > 0
            const initialSeasonNumber = currentPlayingSeasonNumber || 
                                        seasons.find(s => s.season_number > 0)?.season_number || 
                                        seasons[0].season_number;

            const initialSeason = seasons.find(s => s.season_number === initialSeasonNumber);

            // Only update if a valid initial season is found and it's different from the current
            if (initialSeason && initialSeason.id !== selectedSeason?.id) {
                setSelectedSeason(initialSeason);
            }
        }
    // No dependency on selectedSeason here to avoid infinite loop with Effect 3
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [seasons]); 

    // 🌟 Effect 3: SYNCHRONIZATION FIX (Updates selectedSeason when parent state changes via URL)
    useEffect(() => {
        if (currentPlayingSeasonNumber && selectedSeason?.season_number !== currentPlayingSeasonNumber) {
            const seasonToSelect = seasons.find(s => s.season_number === currentPlayingSeasonNumber);
            if (seasonToSelect) {
                setSelectedSeason(seasonToSelect);
            }
        }
    }, [currentPlayingSeasonNumber, seasons, selectedSeason]);


    const availableSeasons = seasons.filter(s => s.season_number !== 0);

    if (!availableSeasons.length) {
        return <div className="text-gray-400">No season information available.</div>;
    }

    const formatSeasonName = (name) => {
        if (!name) return '';
        if (name.length > MAX_SEASON_NAME_LENGTH) {
            return name.slice(0, MAX_SEASON_NAME_LENGTH - 3) + '...';
        }
        return name;
    };

    const handleSeasonChange = (season) => {
        setSelectedSeason(season);
        setIsSeasonListOpen(false); // Close the list after selection
    };

    // 🌟 Enhancement: Memoize the episode click handler for better performance
    const handleEpisodeClick = useCallback((episode) => {
        onEpisodeClick(episode);
    }, [onEpisodeClick]);

    return (
        <div>
            {/* CUSTOM STYLED SEASON SELECTOR */}
            <div className="relative inline-block text-left mb-6">
                <button
                    className="flex items-center justify-between w-full bg-white/10 text-white py-2 pl-4 pr-3 rounded-lg hover:bg-white/20 transition-colors min-w-48"
                    onClick={() => setIsSeasonListOpen(prev => !prev)}
                >
                    <span className="truncate">{selectedSeason ? formatSeasonName(selectedSeason.name) : 'Select Season'}</span>
                    <ChevronDown className={`w-4 h-4 ml-2 transition-transform ${isSeasonListOpen ? 'rotate-180' : 'rotate-0'}`} />
                </button>

                {isSeasonListOpen && (
                    <div className="absolute left-0 mt-1 w-full max-h-60 overflow-y-auto bg-gray-900 border border-pink-600 rounded-lg shadow-lg z-20">
                        {availableSeasons.map((season) => (
                            <div
                                key={season.id}
                                className={`py-2 px-4 cursor-pointer transition-colors ${
                                    selectedSeason && selectedSeason.id === season.id
                                        ? 'bg-pink-600 font-semibold' 
                                        : 'hover:bg-white/10' 
                                }`}
                                onClick={() => handleSeasonChange(season)}
                                title={season.name}
                            >
                                <span className="truncate">
                                    {formatSeasonName(season.name)}
                                </span>
                            </div>
                        ))}
                    </div>
                )}
            </div>

            {loadingEpisodes && <div className="text-pink-500">Loading episodes...</div>}
            {error && <div className="text-red-500">{error}</div>}

            {!loadingEpisodes && !error && (
                <div className="space-y-4">
                    {episodes.map((episode) => (
                        <div 
                            key={episode.id} 
                            className="flex gap-4 p-3  rounded-lg border border-gray-800 hover:bg-white/10 transition cursor-pointer"
                            // Use memoized handler
                            onClick={() => handleEpisodeClick(episode)} 
                        >
                            <div className="relative flex-shrink-0 w-32 h-20 rounded-md overflow-hidden bg-black">
                                <img
                                    src={
                                        episode.still_path
                                            ? `https://image.tmdb.org/t/p/w300${episode.still_path}`
                                            : '/No-Poster.png'
                                    }
                                    alt={`S${episode.season_number} E${episode.episode_number}`}
                                    className="w-full h-full object-cover"
                                />
                                <div className="absolute inset-0 flex items-center justify-center bg-black/40 hover:bg-black/60 transition">
                                    <Play size={24} className="text-white fill-current" />
                                </div>
                            </div>

                            <div className="flex-grow">
                                <h3 className="font-semibold text-xs md:text-xs lg:text-base">
                                    {episode.episode_number}. {episode.name}
                                </h3>
                                <p className="text-xs hidden sm:hidden md:hidden lg:block text-gray-400 mt-1 line-clamp-2">
                                    {episode.overview || 'No description available.'}
                                </p>
                            </div>

                            <div className="flex-shrink-0 text-right">
                                {/* 🌟 Fixed: Added check for episode.runtime to prevent NaN/null errors */}
                                {episode.runtime !== null && episode.runtime !== undefined && (
                                    <p className="text-xs text-gray-400 md:text-xs">
                                        {Math.floor(episode.runtime / 60)}h {episode.runtime % 60}m
                                    </p>
                                )}
                            </div>
                        </div>
                    ))}
                </div>
            )}

             {!loadingEpisodes && !error && episodes.length === 0 && (
                <div className="text-gray-400">No episodes found for this season.</div>
            )}
        </div>
    );
});

        }, [navigate]);
        return null;
    }

    // Initialize state from URL Search Params

    const initialEpisode = parseInt(searchParams.get('e')) || 1;

        episode_number: initialEpisode,
    });

    const tmdbId = isAnilist ? null : movie.id;

    const updatePlayingEpisode = useCallback((season, episode) => {
        setPlayingEpisode({
            season_number: season,
            episode_number: episode,
        });

        // Update URL search parameters
        setSearchParams({ s: season, e: episode }, { replace: true });
    }, [setSearchParams]); 


    // Combined Embed URL Logic - **Uses playingEpisode state** (ALREADY MEMOIZED)
    const getEmbedUrl = useCallback(() => {
        const links = CUSTOM_EMBED_LINKS[movie.id];
        const episodeNum = playingEpisode?.episode_number || 1; // Uses state
        const seasonNum = playingEpisode?.season_number || 1; // Uses state

        if (links) {
            if (typeof links === 'object' && !Array.isArray(links)) {
                const seasonLinks = links[seasonNum];

                if (seasonLinks && Array.isArray(seasonLinks)) {
                    let linkIndex = episodeNum - 1; 

                    if (movie.id === 31910) {
                        const startEpisode = NARUTO_SHIPPUDEN_START_EPISODES[seasonNum] || 1;
                        // For Naruto, linkIndex is the offset from the starting TMDB episode number
                        linkIndex = episodeNum - startEpisode; 
                    }

                    if (linkIndex >= 0 && linkIndex < seasonLinks.length) {
                        let url = seasonLinks[linkIndex];
                        // Normalize Google Drive link for embedding
                        if (url.includes('drive.google.com') && !url.includes('/preview')) {
                            url = url.replace('/view?usp=drivesdk', '/preview');
                            url = url.replace(/\/file\/d\/([\w-]+)\/view/g, '/file/d/$1/preview');
                        }
                        return url;
                    }
                }
            }

            if (Array.isArray(links)) {
                return links[episodeNum - 1] || links[0];
            }

            if (typeof links === 'string') {
                return links;
            }
        }

            return `https://player.vidplus.to/embed/anime/${movie.id}`;
        } else if (isTV) {
            return `https://player.vidplus.to/embed/tv/${movie.id}-${seasonNum}-${episodeNum}`;

            return `https://player.vidplus.to/embed/movie/${movie.id}`;

            updatePlayingEpisode(currentSeasonNum, nextEpisodeNum);

            // Check for the next season
            const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);

                let nextSeasonStartEpisode = 1;

                    nextSeasonStartEpisode = NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number] || 1;
                }
                updatePlayingEpisode(nextSeason.season_number, nextSeasonStartEpisode);

                alert("Congratulations! You've reached the end of the available episodes.");

            }

            // Update state and URL to the first episode of the next season
            updatePlayingEpisode(nextSeason.season_number, nextSeasonStartEpisode);
        } else {
            alert("You've reached the final season.");
        }

        // Filter out Season 0 (Specials) and sort by season number

            let previousSeasonEndEpisode = 1;

                // For Naruto, navigate to the last episode of the previous season
                previousSeasonEndEpisode = NARUTO_SHIPPUDEN_END_EPISODES[previousSeason.season_number] || 1;

                setLogoPath(englishLogo?.file_path || null);

            } catch (err) {
                console.error("Failed to fetch player extra details:", err);
            }
        };
        fetchExtraDetails();
    }, [movie, isAnilist, seasons.length, tmdbId]);


    const title = isAnilist ? movie.title.english || movie.title.romaji : movie.title || movie.name;
    const hasEpisodes = seasons.length > 0 && !isAnilist && isTV; 

    // Logic to determine if there is a next episode available (ALREADY WELL-STRUCTURED)
    const canPlayNext = (() => {
        if (!isTV || !playingEpisode || !currentSeasonDetails) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const currentEpisodeNum = playingEpisode.episode_number;

        let maxEpisodeInCurrentSeason;
        if (movie.id === 31910) {
            maxEpisodeInCurrentSeason = NARUTO_SHIPPUDEN_END_EPISODES[currentSeasonNum];
        } else {
            // Use episodes array from the full season details
            maxEpisodeInCurrentSeason = currentSeasonDetails.episodes ? currentSeasonDetails.episodes.length : 0;
        }

        if (currentEpisodeNum < maxEpisodeInCurrentSeason) {
            return true;
        }

        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
        const nextSeason = allSeasons[currentSeasonIndex + 1];

        // If current episode is the last one, check if there's a next season.
        if (nextSeason) {
            if (movie.id === 31910) {
                // For Naruto, ensure the next season has a defined start episode
                return !!NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number];
            }
            return true;
        }

        return false;
    })();

    // Logic to determine if there is a next season available (ALREADY WELL-STRUCTURED)
    const canSkipToNextSeason = (() => {
        if (!isTV || !playingEpisode) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);

        // Next season exists if the current season is not the last one in the sorted list
        return currentSeasonIndex !== -1 && currentSeasonIndex < allSeasons.length - 1;
    })();

    // 🌟 NEW: Logic to determine if there is a previous season available
    const canSkipToPreviousSeason = (() => {
        if (!isTV || !playingEpisode) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);

        // Previous season exists if the current season is not the *first* one in the sorted list
        return currentSeasonIndex > 0;
    })();


    const logoUrl = logoPath ? `https://image.tmdb.org/t/p/w500${logoPath}` : null;


    // -------------------------------------------------------------------
    // 3. PLAYER SCREEN RENDER
    // -------------------------------------------------------------------
    return (
        <div className="fixed top-0 inset-0 bg-black text-white w-full h-full overflow-y-auto animate-fade-in z-[1000]">
            {/* Main container for the split layout */}
            <div className="flex flex-col md:flex-row h-full"> 

                {/* Player Section */}
                <div className="w-full md:w-2/3 flex-shrink-0"> 
                    <div className="relative w-full sticky top-0 z-50 aspect-video animate-fade-in"> 
                        <iframe
                            src={embedUrl}
                            title="Content Player"
                            allowFullScreen
                            className="w-full h-full"
                            key={embedUrl} // Key forces iframe reload on episode change
                        />

                        {/* Back Button */}
                        <button
                            className="absolute top-4 left-4 bg-black/70 px-4 py-2 rounded-lg hover:bg-red-600 transition flex items-center gap-1 text-sm z-50"
                            onClick={handleBackToDetails} // Memoized handler
                        >
                            <ArrowLeft size={16} /> Back to Details
                        </button>
                    </div>

                    <div className="p-6">
                        <div className="flex justify-between items-center">
                            <div>
                                {logoUrl ? ( 
                                    <img src={logoUrl} 
                                        alt={title}
                                        className="max-h-12 w-auto object-contain" 
                                    /> 
                                ) : (
                                    <h1 className="text-xm font-bold">{title}</h1>
                                )}
                                {playingEpisode && isTV && (
                                    <p className="text-gray-400 mt-1">
                                        Now Playing: S{playingEpisode.season_number}  E{playingEpisode.episode_number}
                                    </p>
                                )}
                            </div>

                            {/* PREV/NEXT SEASON/EPISODE BUTTONS CONTAINER */}
                            <div className="flex gap-4 text-xs">
                                {/* 🌟 NEW: PREVIOUS SEASON BUTTON */}
                                {canSkipToPreviousSeason && (
                                    <button
                                        className="flex items-center gap-2 px-3 py-1 rounded-lg border border-gray-600 text-gray-400 hover:bg-gray-700 hover:text-white transition"

                                    >
                                        <ArrowLeft size={18} /> Prev Season
                                    </button>
                                )}

                                {/* NEXT EPISODE BUTTON */}
                                {canPlayNext && (
                                    <button
                                        className="flex items-center gap-2 px-3 py-1 rounded-lg bg-pink-600 hover:bg-pink-700 transition"
                                        onClick={handleNextEpisode} // Memoized handler
                                    >
                                        Next Episode <SkipForward size={18} />
                                    </button>
                                )}

                                {/* NEXT SEASON BUTTON */}
                                {canSkipToNextSeason && (
                                    <button
                                        className="flex items-center gap-2 px-3 py-1 rounded-lg border border-pink-600 text-pink-500 hover:bg-pink-600 hover:text-white transition"
                                        onClick={handleNextSeason} // Memoized handler
                                    >
                                        Next Season <SkipForward size={18} />
                                    </button>
                                )}
                            </div>
                        </div>
                    </div>
                </div>

                {/* Episodes List/Sidebar Section */}
                {hasEpisodes && (
                    <div className="w-full h-full overflow-y-auto p-4 md:w-1/3 md:overflow-y-auto md:h-full md:border-l md:border-white/10">
                        <h2 className="text-xl font-semibold mb-4 border-b border-white/10 pb-2">Episodes</h2>

                        <EpisodesList className="animate-fade-in"
                            tvId={movie.id} 
                            seasons={seasons} 
                            onEpisodeClick={handleEpisodeClick} // Memoized handler
                            onSeasonSelect={handleSeasonSelect} // Memoized handler
                            // Pass the current playing season number to control the child list
                            currentPlayingSeasonNumber={playingEpisode.season_number} 
                        />
                    </div>
                )}
               </div>
      </div>
    );
};

    7: 144, 8: 152, 10: 197, 12: 243, 13: 276, 
    14: 296, 15: 321,

// 1. EPISODES LIST COMPONENT (Nested Helper)

const EpisodesList = React.memo(({ tvId, seasons, onEpisodeClick, onSeasonSelect, currentPlayingSeasonNumber }) => {
    if (!tvId) return null;

    const [selectedSeason, setSelectedSeason] = useState(null);
    const [episodes, setEpisodes] = useState([]);
    const [loadingEpisodes, setLoadingEpisodes] = useState(false);
    const [error, setError] = useState(null);
    const [isSeasonListOpen, setIsSeasonListOpen] = useState(false); 

    const MAX_SEASON_NAME_LENGTH = 20;

    const fetchEpisodes = useCallback(async (seasonNumber) => {
        if (!tvId || !seasonNumber) return;

        setLoadingEpisodes(true);
        setError(null);
        try {
            const res = await fetch(
                `${API_BASE_URL}/tv/${tvId}/season/${seasonNumber}?api_key=${API_KEY}`
            );
            const data = await res.json();

            if (data.success === false) {
                setError("Failed to load episodes for this season.");
                setEpisodes([]);
                const currentSeasonData = seasons.find(s => s.season_number === seasonNumber) || null;
                onSeasonSelect(currentSeasonData);
                return;
            }

            setEpisodes(data.episodes || []);
            // Pass the entire season details object back to the parent
            onSeasonSelect(data); 

        } catch (err) {
            setError("Error fetching episodes.");
        } finally {
            setLoadingEpisodes(false);
        }
    }, [tvId, seasons, onSeasonSelect]);

    // Effect 1: Fetch episodes whenever selectedSeason changes
    useEffect(() => {
        if (selectedSeason) {
            fetchEpisodes(selectedSeason.season_number);
        }
    }, [selectedSeason, fetchEpisodes]);

    // Effect 2: Set initial season on mount or when seasons change
    useEffect(() => {
        if (seasons && seasons.length > 0) {
            // Priority: Use the season number from the URL/parent if available, otherwise default to first season > 0
            const initialSeasonNumber = currentPlayingSeasonNumber || 
                                        seasons.find(s => s.season_number > 0)?.season_number || 
                                        seasons[0].season_number;

            const initialSeason = seasons.find(s => s.season_number === initialSeasonNumber);

            // Only update if a valid initial season is found and it's different from the current
            if (initialSeason && initialSeason.id !== selectedSeason?.id) {
                setSelectedSeason(initialSeason);
            }
        }
    // No dependency on selectedSeason here to avoid infinite loop with Effect 3
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [seasons]); 

    // Effect 3: SYNCHRONIZATION FIX (Updates selectedSeason when parent state changes via URL)
    useEffect(() => {
        if (currentPlayingSeasonNumber && selectedSeason?.season_number !== currentPlayingSeasonNumber) {
            const seasonToSelect = seasons.find(s => s.season_number === currentPlayingSeasonNumber);
            if (seasonToSelect) {
                setSelectedSeason(seasonToSelect);
            }
        }
    }, [currentPlayingSeasonNumber, seasons, selectedSeason]);


    const availableSeasons = seasons.filter(s => s.season_number !== 0);

    if (!availableSeasons.length) {
        return <div className="text-gray-400">No season information available.</div>;
    }

    const formatSeasonName = (name) => {
        if (!name) return '';
        if (name.length > MAX_SEASON_NAME_LENGTH) {
            return name.slice(0, MAX_SEASON_NAME_LENGTH - 3) + '...';
        }
        return name;
    };

    const handleSeasonChange = (season) => {
        setSelectedSeason(season);
        setIsSeasonListOpen(false); // Close the list after selection
    };

    // Memoize the episode click handler for better performance
    const handleEpisodeClick = useCallback((episode) => {
        onEpisodeClick(episode);
    }, [onEpisodeClick]);

    return (
        <div>
            {/* CUSTOM STYLED SEASON SELECTOR */}
            <div className="relative inline-block text-left mb-6">
                <button
                    className="flex items-center justify-between w-full bg-white/10 text-white py-2 pl-4 pr-3 rounded-lg hover:bg-white/20 transition-colors min-w-48"
                    onClick={() => setIsSeasonListOpen(prev => !prev)}
                >
                    <span className="truncate">{selectedSeason ? formatSeasonName(selectedSeason.name) : 'Select Season'}</span>
                    <ChevronDown className={`w-4 h-4 ml-2 transition-transform ${isSeasonListOpen ? 'rotate-180' : 'rotate-0'}`} />
                </button>

                {isSeasonListOpen && (
                    <div className="absolute left-0 mt-1 w-full max-h-60 overflow-y-auto bg-gray-900 border border-pink-600 rounded-lg shadow-lg z-20">
                        {availableSeasons.map((season) => (
                            <div
                                key={season.id}
                                className={`py-2 px-4 cursor-pointer transition-colors ${
                                    selectedSeason && selectedSeason.id === season.id
                                        ? 'bg-pink-600 font-semibold' 
                                        : 'hover:bg-white/10' 
                                }`}
                                onClick={() => handleSeasonChange(season)}
                                title={season.name}
                            >
                                <span className="truncate">
                                    {formatSeasonName(season.name)}
                                </span>
                            </div>
                        ))}
                    </div>
                )}
            </div>

            {loadingEpisodes && <div className="text-pink-500">Loading episodes...</div>}
            {error && <div className="text-red-500">{error}</div>}

            {!loadingEpisodes && !error && (
                <div className="space-y-4">
                    {episodes.map((episode) => (
                        <div 
                            key={episode.id} 
                            className="flex gap-4 p-3  rounded-lg border border-gray-800 hover:bg-white/10 transition cursor-pointer"
                            // Use memoized handler
                            onClick={() => handleEpisodeClick(episode)} 
                        >
                            <div className="relative flex-shrink-0 w-32 h-20 rounded-md overflow-hidden bg-black">
                                <img
                                    src={
                                        episode.still_path
                                            ? `https://image.tmdb.org/t/p/w300${episode.still_path}`
                                            : '/No-Poster.png'
                                    }
                                    alt={`S${episode.season_number} E${episode.episode_number}`}
                                    className="w-full h-full object-cover"
                                />
                                <div className="absolute inset-0 flex items-center justify-center bg-black/40 hover:bg-black/60 transition">
                                    <Play size={24} className="text-white fill-current" />
                                </div>
                            </div>

                            <div className="flex-grow">
                                <h3 className="font-semibold text-xs md:text-xs lg:text-base">
                                    {episode.episode_number}. {episode.name}
                                </h3>
                                <p className="text-xs hidden sm:hidden md:hidden lg:block text-gray-400 mt-1 line-clamp-2">
                                    {episode.overview || 'No description available.'}
                                </p>
                            </div>

                            <div className="flex-shrink-0 text-right">
                                {/* Fixed: Added check for episode.runtime to prevent NaN/null errors */}
                                {episode.runtime !== null && episode.runtime !== undefined && (
                                    <p className="text-xs text-gray-400 md:text-xs">
                                        {Math.floor(episode.runtime / 60)}h {episode.runtime % 60}m
                                    </p>
                                )}
                            </div>
                        </div>
                    ))}
                </div>
            )}

             {!loadingEpisodes && !error && episodes.length === 0 && (
                <div className="text-gray-400">No episodes found for this season.</div>
            )}
        </div>
    );
});

const DedicatedPlayerScreen = ({ movie, isTV }) => { 

    const navigate = useNavigate();
    const [searchParams, setSearchParams] = useSearchParams(); 

    // Guard Clause
    if (!movie) {
        useEffect(() => {
            navigate("/", { replace: true });
        }, [navigate]);
        return null;
    }

    // Initialize state from URL Search Params
    const initialSeason = parseInt(searchParams.get('s')) || 1;
    const initialEpisode = parseInt(searchParams.get('e')) || 1;

    const [seasons, setSeasons] = useState(movie.seasons || []);
    // currentSeasonDetails will hold the full season object (including episodes array) from TMDB API
    const [currentSeasonDetails, setCurrentSeasonDetails] = useState(null); 
    const [playingEpisode, setPlayingEpisode] = useState({
        season_number: initialSeason,
        episode_number: initialEpisode,
    });
    const [logoPath, setLogoPath] = useState(null);

    const isAnilist = movie.media_type === 'anilist' || !!movie?.title?.romaji;
    const tmdbId = isAnilist ? null : movie.id; 

    // Helper to update the episode state AND the URL search parameters
    const updatePlayingEpisode = useCallback((season, episode) => {
        setPlayingEpisode({
            season_number: season,
            episode_number: episode,
        });

        // Update URL search parameters
        setSearchParams({ s: season, e: episode }, { replace: true });
    }, [setSearchParams]); 


    // Combined Embed URL Logic - **Uses playingEpisode state**
    const getEmbedUrl = useCallback(() => {
        const links = CUSTOM_EMBED_LINKS[movie.id];
        const episodeNum = playingEpisode?.episode_number || 1; // Uses state
        const seasonNum = playingEpisode?.season_number || 1; // Uses state

        if (links) {
            if (typeof links === 'object' && !Array.isArray(links)) {
                const seasonLinks = links[seasonNum];

                if (seasonLinks && Array.isArray(seasonLinks)) {
                    let linkIndex = episodeNum - 1; 

                    if (movie.id === 31910) {
                        const startEpisode = NARUTO_SHIPPUDEN_START_EPISODES[seasonNum] || 1;
                        // For Naruto, linkIndex is the offset from the starting TMDB episode number
                        linkIndex = episodeNum - startEpisode; 
                    }

                    if (linkIndex >= 0 && linkIndex < seasonLinks.length) {
                        let url = seasonLinks[linkIndex];
                        // Normalize Google Drive link for embedding
                        if (url.includes('drive.google.com') && !url.includes('/preview')) {
                            url = url.replace('/view?usp=drivesdk', '/preview');
                            url = url.replace(/\/file\/d\/([\w-]+)\/view/g, '/file/d/$1/preview');
                        }
                        return url;
                    }
                }
            }

            if (Array.isArray(links)) {
                return links[episodeNum - 1] || links[0];
            }

            if (typeof links === 'string') {
                return links;
            }
        }

        // Fallback to generic embed provider (Vidplus.to)
        if (isAnilist) {
            return `https://player.vidplus.to/embed/anime/${movie.id}`;
        } else if (isTV) {
            return `https://player.vidplus.to/embed/tv/${movie.id}-${seasonNum}-${episodeNum}`; 
        } else {
            return `https://player.vidplus.to/embed/movie/${movie.id}`;
        }
    }, [movie.id, isAnilist, isTV, playingEpisode]);

    const embedUrl = getEmbedUrl();

    // Handle full season details coming back from EpisodesList component
    const handleSeasonSelect = useCallback((seasonDetails) => {
        setCurrentSeasonDetails(seasonDetails);
    }, []);

    // **MEMOIZED** Episode Navigation Logic (Next Episode)
    const handleNextEpisode = useCallback(() => {
        if (!playingEpisode || !isTV || !currentSeasonDetails) return;

        const currentSeasonNum = playingEpisode.season_number;
        const currentEpisodeNum = playingEpisode.episode_number;
        const nextEpisodeNum = currentEpisodeNum + 1;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);

        let maxEpisodeInCurrentSeason;
        if (movie.id === 31910) {
            maxEpisodeInCurrentSeason = NARUTO_SHIPPUDEN_END_EPISODES[currentSeasonNum];
        } else {
            // Use the episode count from the currentSeasonDetails API response
            maxEpisodeInCurrentSeason = currentSeasonDetails.episodes ? currentSeasonDetails.episodes.length : 0;
        }

        if (nextEpisodeNum <= maxEpisodeInCurrentSeason) {
            // Play next episode in the same season
            updatePlayingEpisode(currentSeasonNum, nextEpisodeNum); 
        } else {
            // Check for the next season
            const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
            const nextSeason = allSeasons[currentSeasonIndex + 1];

            if (nextSeason) {
                let nextSeasonStartEpisode = 1; 
                if (movie.id === 31910) {
                    nextSeasonStartEpisode = NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number] || 1;
                }
                updatePlayingEpisode(nextSeason.season_number, nextSeasonStartEpisode); 
            } else {
                alert("Congratulations! You've reached the end of the available episodes.");
            }
        }
    }, [playingEpisode, isTV, currentSeasonDetails, seasons, movie.id, updatePlayingEpisode]);

    // Handle jump to next season
    const handleNextSeason = useCallback(() => {
        if (!isTV || !playingEpisode) return;

        const currentSeasonNum = playingEpisode.season_number;
        // Filter out Season 0 (Specials) and sort by season number
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
        const nextSeason = allSeasons[currentSeasonIndex + 1];

        if (nextSeason) {
            let nextSeasonStartEpisode = 1; 
            if (movie.id === 31910) {
                nextSeasonStartEpisode = NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number] || 1;
            }

            // Update state and URL to the first episode of the next season
            updatePlayingEpisode(nextSeason.season_number, nextSeasonStartEpisode);
        } else {
            alert("You've reached the final season.");
        }
    }, [isTV, playingEpisode, seasons, movie.id, updatePlayingEpisode]); 

    // 🌟 NEW: Handle jump to previous season
    const handlePreviousSeason = useCallback(() => {
        if (!isTV || !playingEpisode) return;

        const currentSeasonNum = playingEpisode.season_number;
        // Filter out Season 0 (Specials) and sort by season number
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
        const previousSeason = allSeasons[currentSeasonIndex - 1]; // Look one index back

        if (previousSeason) {
            let targetEpisode = 1; 
            if (movie.id === 31910) {
                // For Naruto, navigate to the last episode of the previous season
                targetEpisode = NARUTO_SHIPPUDEN_END_EPISODES[previousSeason.season_number] || 1;
            }
            // For others, we default to episode 1 of the previous season.

            // Update state and URL to the target episode of the previous season
            updatePlayingEpisode(previousSeason.season_number, targetEpisode);
        } else {
            alert("You are already on the first season.");
        }
    }, [isTV, playingEpisode, seasons, movie.id, updatePlayingEpisode]);


    // Update handleEpisodeClick to use the new helper and update URL
    const handleEpisodeClick = useCallback((episode) => {
        updatePlayingEpisode(episode.season_number, episode.episode_number);
    }, [updatePlayingEpisode]);

    // Memoize handleBackToDetails
    const handleBackToDetails = useCallback(() => {
        navigate(-1); 
    }, [navigate]);

    // Fetch details for seasons/logo if not already available
    useEffect(() => {
        if (isAnilist || !tmdbId) return;

        const fetchExtraDetails = async () => {
            try {
                // Ensure correct media type is used for API call
                const type = movie.media_type === 'tv' ? "tv" : "movie"; 
                const res = await fetch(
                `${API_BASE_URL}/${type}/${movie.id}?api_key=${API_KEY}&append_to_response=images,seasons`
                );
                const data = await res.json();

                if (data.seasons?.length && !seasons.length) {
                    setSeasons(data.seasons);
                }

                const logos = data.images?.logos || [];
                const englishLogo = logos.find(logo => logo.iso_639_1 === 'en') || logos[0];
                setLogoPath(englishLogo?.file_path || null);

            } catch (err) {
                console.error("Failed to fetch player extra details:", err);
            }
        };
        fetchExtraDetails();
    }, [movie, isAnilist, seasons.length, tmdbId]);


    const title = isAnilist ? movie.title.english || movie.title.romaji : movie.title || movie.name;
    const hasEpisodes = seasons.length > 0 && !isAnilist && isTV; 

    // Logic to determine if there is a next episode available
    const canPlayNext = (() => {
        if (!isTV || !playingEpisode || !currentSeasonDetails) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const currentEpisodeNum = playingEpisode.episode_number;

        let maxEpisodeInCurrentSeason;
        if (movie.id === 31910) {
            maxEpisodeInCurrentSeason = NARUTO_SHIPPUDEN_END_EPISODES[currentSeasonNum];
        } else {
            // Use episodes array from the full season details
            maxEpisodeInCurrentSeason = currentSeasonDetails.episodes ? currentSeasonDetails.episodes.length : 0;
        }

        if (currentEpisodeNum < maxEpisodeInCurrentSeason) {
            return true;
        }

        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
        const nextSeason = allSeasons[currentSeasonIndex + 1];

        // If current episode is the last one, check if there's a next season.
        if (nextSeason) {
            if (movie.id === 31910) {
                // For Naruto, ensure the next season has a defined start episode
                return !!NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number];
            }
            return true;
        }

        return false;
    })();

    // Logic to determine if there is a next season available
    const canSkipToNextSeason = (() => {
        if (!isTV || !playingEpisode) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);

        // Next season exists if the current season is not the last one in the sorted list
        return currentSeasonIndex !== -1 && currentSeasonIndex < allSeasons.length - 1;
    })();

    // Logic to determine if there is a previous season available
    const canSkipToPreviousSeason = (() => {
        if (!isTV || !playingEpisode) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);

        // Previous season exists if the current season is not the *first* one in the sorted list
        return currentSeasonIndex > 0;
    })();


    const logoUrl = logoPath ? `https://image.tmdb.org/t/p/w500${logoPath}` : null;


    // -------------------------------------------------------------------
    // 3. PLAYER SCREEN RENDER
    // -------------------------------------------------------------------
    return (
        <div className="fixed top-0 inset-0 bg-black text-white w-full h-full overflow-y-auto animate-fade-in z-[1000]">
            {/* Main container for the split layout */}
            <div className="flex flex-col md:flex-row h-full"> 

                {/* Player Section */}
                <div className="w-full md:w-2/3 flex-shrink-0"> 
                    <div className="relative w-full sticky top-0 z-50 aspect-video animate-fade-in"> 
                        <iframe
                            src={embedUrl}
                            title="Content Player"
                            allowFullScreen
                            className="w-full h-full"
                            key={embedUrl} // Key forces iframe reload on episode change
                        />

                        {/* Back Button */}
                        <button
                            className="absolute top-4 left-4 bg-black/70 px-4 py-2 rounded-lg hover:bg-red-600 transition flex items-center gap-1 text-sm z-50"
                            onClick={handleBackToDetails} // Memoized handler
                        >
                            <ArrowLeft size={16} /> Back to Details
                        </button>
                    </div>

                    <div className="p-6">
                        {/* FIX: Use flex-col on mobile, flex-row on sm screens and up */}
                        <div className="flex flex-col sm:flex-row sm:justify-between sm:items-center gap-4"> 
                            
                            {/* TITLE / INFO SECTION */}
                            <div>
                                {logoUrl ? ( 
                                    <img src={logoUrl} 
                                        alt={title}
                                        className="max-h-12 w-auto object-contain" 
                                    /> 
                                ) : (
                                    <h1 className="text-xm font-bold">{title}</h1>
                                )}
                                {playingEpisode && isTV && (
                                    <p className="text-gray-400 mt-1">
                                        Now Playing: S{playingEpisode.season_number}  E{playingEpisode.episode_number}
                                    </p>
                                )}
                            </div>

                            {/* PREV/NEXT SEASON/EPISODE BUTTONS CONTAINER */}
                            <div className="flex flex-wrap gap-4 text-xs mt-2 sm:mt-0"> 
                                {/* 🌟 PREVIOUS SEASON BUTTON */}
                                {canSkipToPreviousSeason && (
                                    <button
                                        className="flex items-center gap-2 px-3 py-1 rounded-lg border border-gray-600 text-gray-400 hover:bg-gray-700 hover:text-white transition"
                                        onClick={handlePreviousSeason} // Memoized handler
                                    >
                                        <ArrowLeft size={18} /> Prev Season
                                    </button>
                                )}

                                {/* NEXT EPISODE BUTTON */}
                                {canPlayNext && (
                                    <button
                                        className="flex items-center gap-2 px-3 py-1 rounded-lg bg-pink-600 hover:bg-pink-700 transition"
                                        onClick={handleNextEpisode} // Memoized handler
                                    >
                                        Next Episode <SkipForward size={18} />
                                    </button>
                                )}

                                {/* NEXT SEASON BUTTON */}
                                {canSkipToNextSeason && (
                                    <button
                                        className="flex items-center gap-2 px-3 py-1 rounded-lg border border-pink-600 text-pink-500 hover:bg-pink-600 hover:text-white transition"
                                        onClick={handleNextSeason} // Memoized handler
                                    >
                                        Next Season <SkipForward size={18} />
                                    </button>
                                )}
                            </div>
                        </div>
                    </div>
                </div>

                {/* Episodes List/Sidebar Section */}
                {hasEpisodes && (
                    <div className="w-full h-full overflow-y-auto p-4 md:w-1/3 md:overflow-y-auto md:h-full md:border-l md:border-white/10">
                        <h2 className="text-xl font-semibold mb-4 border-b border-white/10 pb-2">Episodes</h2>

                        <EpisodesList className="animate-fade-in"
                            tvId={movie.id} 
                            seasons={seasons} 
                            onEpisodeClick={handleEpisodeClick} // Memoized handler
                            onSeasonSelect={handleSeasonSelect} // Memoized handler
                            // Pass the current playing season number to control the child list
                            currentPlayingSeasonNumber={playingEpisode.season_number} 
                        />
                    </div>
                )}
               </div>
      </div>
    );
};

    7: 144, 8: 152, 10: 197, 12: 243, 13: 276, 
    14: 296, 15: 321,

// 1. EPISODES LIST COMPONENT (Nested Helper)

const EpisodesList = React.memo(({ 
    tvId, 
    seasons, 
    onEpisodeClick, 
    onSeasonSelect, 
    currentPlayingSeasonNumber,
    currentPlayingEpisodeNumber // <--- ADDED PROP
}) => {
    if (!tvId) return null;

    const [selectedSeason, setSelectedSeason] = useState(null);
    const [episodes, setEpisodes] = useState([]);
    const [loadingEpisodes, setLoadingEpisodes] = useState(false);
    const [error, setError] = useState(null);
    const [isSeasonListOpen, setIsSeasonListOpen] = useState(false); 

    const MAX_SEASON_NAME_LENGTH = 20;

    const fetchEpisodes = useCallback(async (seasonNumber) => {
        if (!tvId || !seasonNumber) return;

        setLoadingEpisodes(true);
        setError(null);
        try {
            const res = await fetch(
                `${API_BASE_URL}/tv/${tvId}/season/${seasonNumber}?api_key=${API_KEY}`
            );
            const data = await res.json();

            if (data.success === false) {
                setError("Failed to load episodes for this season.");
                setEpisodes([]);
                const currentSeasonData = seasons.find(s => s.season_number === seasonNumber) || null;
                onSeasonSelect(currentSeasonData);
                return;
            }

            setEpisodes(data.episodes || []);
            // Pass the entire season details object back to the parent
            onSeasonSelect(data); 

        } catch (err) {
            setError("Error fetching episodes.");
        } finally {
            setLoadingEpisodes(false);
        }
    }, [tvId, seasons, onSeasonSelect]);

    // Effect 1: Fetch episodes whenever selectedSeason changes
    useEffect(() => {
        if (selectedSeason) {
            fetchEpisodes(selectedSeason.season_number);
        }
    }, [selectedSeason, fetchEpisodes]);

    // Effect 2: Set initial season on mount or when seasons change
    useEffect(() => {
        if (seasons && seasons.length > 0) {
            // Priority: Use the season number from the URL/parent if available, otherwise default to first season > 0
            const initialSeasonNumber = currentPlayingSeasonNumber || 
                                        seasons.find(s => s.season_number > 0)?.season_number || 
                                        seasons[0].season_number;

            const initialSeason = seasons.find(s => s.season_number === initialSeasonNumber);

            // Only update if a valid initial season is found and it's different from the current
            if (initialSeason && initialSeason.id !== selectedSeason?.id) {
                setSelectedSeason(initialSeason);
            }
        }
    // No dependency on selectedSeason here to avoid infinite loop with Effect 3
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [seasons]); 

    // Effect 3: SYNCHRONIZATION FIX (Updates selectedSeason when parent state changes via URL)
    useEffect(() => {
        if (currentPlayingSeasonNumber && selectedSeason?.season_number !== currentPlayingSeasonNumber) {
            const seasonToSelect = seasons.find(s => s.season_number === currentPlayingSeasonNumber);
            if (seasonToSelect) {
                setSelectedSeason(seasonToSelect);
            }
        }
    }, [currentPlayingSeasonNumber, seasons, selectedSeason]);


    const availableSeasons = seasons.filter(s => s.season_number !== 0);

    if (!availableSeasons.length) {
        return <div className="text-gray-400">No season information available.</div>;
    }

    const formatSeasonName = (name) => {
        if (!name) return '';
        if (name.length > MAX_SEASON_NAME_LENGTH) {
            return name.slice(0, MAX_SEASON_NAME_LENGTH - 3) + '...';
        }
        return name;
    };

    const handleSeasonChange = (season) => {
        setSelectedSeason(season);
        setIsSeasonListOpen(false); // Close the list after selection
    };

    // Memoize the episode click handler for better performance
    const handleEpisodeClick = useCallback((episode) => {
        onEpisodeClick(episode);
    }, [onEpisodeClick]);

    return (
        <div>
            {/* CUSTOM STYLED SEASON SELECTOR */}
            <div className="relative inline-block text-left mb-6">
                <button
                    className="flex items-center justify-between w-full bg-white/10 text-white py-2 pl-4 pr-3 rounded-lg hover:bg-white/20 transition-colors min-w-48"
                    onClick={() => setIsSeasonListOpen(prev => !prev)}
                >
                    <span className="truncate">{selectedSeason ? formatSeasonName(selectedSeason.name) : 'Select Season'}</span>
                    <ChevronDown className={`w-4 h-4 ml-2 transition-transform ${isSeasonListOpen ? 'rotate-180' : 'rotate-0'}`} />
                </button>

                {isSeasonListOpen && (
                    <div className="absolute left-0 mt-1 w-full max-h-60 overflow-y-auto bg-gray-900 border border-pink-600 rounded-lg shadow-lg z-20">
                        {availableSeasons.map((season) => (
                            <div
                                key={season.id}
                                className={`py-2 px-4 cursor-pointer transition-colors ${
                                    selectedSeason && selectedSeason.id === season.id
                                        ? 'bg-pink-600 font-semibold' 
                                        : 'hover:bg-white/10' 
                                }`}
                                onClick={() => handleSeasonChange(season)}
                                title={season.name}
                            >
                                <span className="truncate">
                                    {formatSeasonName(season.name)}
                                </span>
                            </div>
                        ))}
                    </div>
                )}
            </div>

            {loadingEpisodes && <div className="text-pink-500">Loading episodes...</div>}
            {error && <div className="text-red-500">{error}</div>}

            {!loadingEpisodes && !error && (
                <div className="space-y-4">
                    {episodes.map((episode) => {
                        const isCurrentEpisode = 
                            episode.season_number === currentPlayingSeasonNumber && 
                            episode.episode_number === currentPlayingEpisodeNumber;

                        return (
                            <div 
                                key={episode.id} 
                                // UPDATED STYLING: Minimal background, border for active episode
                                className={`
                                    flex gap-4 p-3 rounded-lg transition cursor-pointer 
                                    ${isCurrentEpisode 
                                        ? 'bg-pink-600/20 border border-pink-600' // Highlight active episode
                                        : 'bg-white/5 hover:bg-white/10 border border-transparent' // Default style
                                    }
                                `}
                                // Use memoized handler
                                onClick={() => handleEpisodeClick(episode)} 
                            >
                                {/* Thumbnail Section */}
                                <div className="relative flex-shrink-0 w-32 h-20 rounded-md overflow-hidden bg-black">
                                    <img
                                        src={
                                            episode.still_path
                                                ? `https://image.tmdb.org/t/p/w300${episode.still_path}`
                                                : '/No-Poster.png'
                                        }
                                        alt={`S${episode.season_number} E${episode.episode_number}`}
                                        className="w-full h-full object-cover"
                                    />
                                    {/* Play icon REMOVED to match image style */}
                                </div>

                                {/* Text Content Section */}
                                <div className="flex-grow flex flex-col justify-center">
                                    {/* Episode Number and Title */}
                                    <h3 className="font-semibold text-base">
                                        {episode.episode_number}. {episode.name}
                                    </h3>
                                    
                                    {/* Date Text (mimicking image style) */}
                                    <p className="text-sm text-gray-400 mt-0">
                                        {episode.air_date 
                                            ? new Date(episode.air_date).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' }) 
                                            : 'Date N/A'}
                                    </p>
                                    
                                    {/* Overview REMOVED to match image style */}
                                </div>

                                {/* Runtime REMOVED to match image style */}
                                {/* The original runtime div is removed here */}
                            </div>
                        );
                    })}
                </div>
            )}

             {!loadingEpisodes && !error && episodes.length === 0 && (
                <div className="text-gray-400">No episodes found for this season.</div>
            )}
        </div>
    );
});

const DedicatedPlayerScreen = ({ movie, isTV }) => { 

    const navigate = useNavigate();
    const [searchParams, setSearchParams] = useSearchParams(); 

    // Guard Clause
    if (!movie) {
        useEffect(() => {
            navigate("/", { replace: true });
        }, [navigate]);
        return null;
    }

    // Initialize state from URL Search Params
    const initialSeason = parseInt(searchParams.get('s')) || 1;
    const initialEpisode = parseInt(searchParams.get('e')) || 1;

    const [seasons, setSeasons] = useState(movie.seasons || []);
    // currentSeasonDetails will hold the full season object (including episodes array) from TMDB API
    const [currentSeasonDetails, setCurrentSeasonDetails] = useState(null); 
    const [playingEpisode, setPlayingEpisode] = useState({
        season_number: initialSeason,
        episode_number: initialEpisode,
    });
    const [logoPath, setLogoPath] = useState(null);

    const isAnilist = movie.media_type === 'anilist' || !!movie?.title?.romaji;
    const tmdbId = isAnilist ? null : movie.id; 

    // Helper to update the episode state AND the URL search parameters
    const updatePlayingEpisode = useCallback((season, episode) => {
        setPlayingEpisode({
            season_number: season,
            episode_number: episode,
        });

        // Update URL search parameters
        setSearchParams({ s: season, e: episode }, { replace: true });
    }, [setSearchParams]); 


    // Combined Embed URL Logic - **Uses playingEpisode state**
    const getEmbedUrl = useCallback(() => {
        const links = CUSTOM_EMBED_LINKS[movie.id];
        const episodeNum = playingEpisode?.episode_number || 1; // Uses state
        const seasonNum = playingEpisode?.season_number || 1; // Uses state

        if (links) {
            if (typeof links === 'object' && !Array.isArray(links)) {
                const seasonLinks = links[seasonNum];

                if (seasonLinks && Array.isArray(seasonLinks)) {
                    let linkIndex = episodeNum - 1; 

                    if (movie.id === 31910) {
                        const startEpisode = NARUTO_SHIPPUDEN_START_EPISODES[seasonNum] || 1;
                        // For Naruto, linkIndex is the offset from the starting TMDB episode number
                        linkIndex = episodeNum - startEpisode; 
                    }

                    if (linkIndex >= 0 && linkIndex < seasonLinks.length) {
                        let url = seasonLinks[linkIndex];
                        // Normalize Google Drive link for embedding
                        if (url.includes('drive.google.com') && !url.includes('/preview')) {
                            url = url.replace('/view?usp=drivesdk', '/preview');
                            url = url.replace(/\/file\/d\/([\w-]+)\/view/g, '/file/d/$1/preview');
                        }
                        return url;
                    }
                }
            }

            if (Array.isArray(links)) {
                return links[episodeNum - 1] || links[0];
            }

            if (typeof links === 'string') {
                return links;
            }
        }

        // Fallback to generic embed provider (Vidplus.to)
        if (isAnilist) {
            return `https://player.vidplus.to/embed/anime/${movie.id}`;
        } else if (isTV) {
            return `https://player.vidplus.to/embed/tv/${movie.id}-${seasonNum}-${episodeNum}`; 
        } else {
            return `https://player.vidplus.to/embed/movie/${movie.id}`;
        }
    }, [movie.id, isAnilist, isTV, playingEpisode]);

    const embedUrl = getEmbedUrl();

    // Handle full season details coming back from EpisodesList component
    const handleSeasonSelect = useCallback((seasonDetails) => {
        setCurrentSeasonDetails(seasonDetails);
    }, []);

    // **MEMOIZED** Episode Navigation Logic (Next Episode)
    const handleNextEpisode = useCallback(() => {
        if (!playingEpisode || !isTV || !currentSeasonDetails) return;

        const currentSeasonNum = playingEpisode.season_number;
        const currentEpisodeNum = playingEpisode.episode_number;
        const nextEpisodeNum = currentEpisodeNum + 1;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);

        let maxEpisodeInCurrentSeason;
        if (movie.id === 31910) {
            maxEpisodeInCurrentSeason = NARUTO_SHIPPUDEN_END_EPISODES[currentSeasonNum];
        } else {
            // Use the episode count from the currentSeasonDetails API response
            maxEpisodeInCurrentSeason = currentSeasonDetails.episodes ? currentSeasonDetails.episodes.length : 0;
        }

        if (nextEpisodeNum <= maxEpisodeInCurrentSeason) {
            // Play next episode in the same season
            updatePlayingEpisode(currentSeasonNum, nextEpisodeNum); 
        } else {
            // Check for the next season
            const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
            const nextSeason = allSeasons[currentSeasonIndex + 1];

            if (nextSeason) {
                let nextSeasonStartEpisode = 1; 
                if (movie.id === 31910) {
                    nextSeasonStartEpisode = NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number] || 1;
                }
                updatePlayingEpisode(nextSeason.season_number, nextSeasonStartEpisode); 
            } else {
                alert("Congratulations! You've reached the end of the available episodes.");
            }
        }
    }, [playingEpisode, isTV, currentSeasonDetails, seasons, movie.id, updatePlayingEpisode]);

    // Handle jump to next season
    const handleNextSeason = useCallback(() => {
        if (!isTV || !playingEpisode) return;

        const currentSeasonNum = playingEpisode.season_number;
        // Filter out Season 0 (Specials) and sort by season number
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
        const nextSeason = allSeasons[currentSeasonIndex + 1];

        if (nextSeason) {
            let nextSeasonStartEpisode = 1; 
            if (movie.id === 31910) {
                nextSeasonStartEpisode = NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number] || 1;
            }

            // Update state and URL to the first episode of the next season
            updatePlayingEpisode(nextSeason.season_number, nextSeasonStartEpisode);
        } else {
            alert("You've reached the final season.");
        }
    }, [isTV, playingEpisode, seasons, movie.id, updatePlayingEpisode]); 

    // 🌟 NEW: Handle jump to previous season
    const handlePreviousSeason = useCallback(() => {
        if (!isTV || !playingEpisode) return;

        const currentSeasonNum = playingEpisode.season_number;
        // Filter out Season 0 (Specials) and sort by season number
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
        const previousSeason = allSeasons[currentSeasonIndex - 1]; // Look one index back

        if (previousSeason) {
            let targetEpisode = 1; 
            if (movie.id === 31910) {
                // For Naruto, navigate to the last episode of the previous season
                targetEpisode = NARUTO_SHIPPUDEN_END_EPISODES[previousSeason.season_number] || 1;
            }
            // For others, we default to episode 1 of the previous season.

            // Update state and URL to the target episode of the previous season
            updatePlayingEpisode(previousSeason.season_number, targetEpisode);
        } else {
            alert("You are already on the first season.");
        }
    }, [isTV, playingEpisode, seasons, movie.id, updatePlayingEpisode]);


    // Update handleEpisodeClick to use the new helper and update URL
    const handleEpisodeClick = useCallback((episode) => {
        updatePlayingEpisode(episode.season_number, episode.episode_number);
    }, [updatePlayingEpisode]);

    // Memoize handleBackToDetails
    const handleBackToDetails = useCallback(() => {
        navigate(-1); 
    }, [navigate]);

    // Fetch details for seasons/logo if not already available
    useEffect(() => {
        if (isAnilist || !tmdbId) return;

        const fetchExtraDetails = async () => {
            try {
                // Ensure correct media type is used for API call
                const type = movie.media_type === 'tv' ? "tv" : "movie"; 
                const res = await fetch(
                `${API_BASE_URL}/${type}/${movie.id}?api_key=${API_KEY}&append_to_response=images,seasons`
                );
                const data = await res.json();

                if (data.seasons?.length && !seasons.length) {
                    setSeasons(data.seasons);
                }

                const logos = data.images?.logos || [];
                const englishLogo = logos.find(logo => logo.iso_639_1 === 'en') || logos[0];
                setLogoPath(englishLogo?.file_path || null);

            } catch (err) {
                console.error("Failed to fetch player extra details:", err);
            }
        };
        fetchExtraDetails();
    }, [movie, isAnilist, seasons.length, tmdbId]);


    const title = isAnilist ? movie.title.english || movie.title.romaji : movie.title || movie.name;
    const hasEpisodes = seasons.length > 0 && !isAnilist && isTV; 

    // Logic to determine if there is a next episode available
    const canPlayNext = (() => {
        if (!isTV || !playingEpisode || !currentSeasonDetails) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const currentEpisodeNum = playingEpisode.episode_number;

        let maxEpisodeInCurrentSeason;

            maxEpisodeInCurrentSeason = NARUTO_SHIPPUDEN_END_EPISODES[currentSeasonNum];
        } else {
            // Use episodes array from the full season details

        // If current episode is the last one, check if there's a next season.

            if (movie.id === 31910) {
                // For Naruto, ensure the next season has a defined start episode
                return !!NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number];
            }
            return true;
        }

        return false;
    })();

    // Logic to determine if there is a next season available
    const canSkipToNextSeason = (() => {
        if (!isTV || !playingEpisode) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);

        // Next season exists if the current season is not the last one in the sorted list
        return currentSeasonIndex !== -1 && currentSeasonIndex < allSeasons.length - 1;
    })();

    // Logic to determine if there is a previous season available

        if (!isTV || !playingEpisode) return false;

        const currentSeasonNum = playingEpisode.season_number;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);

        // Previous season exists if the current season is not the *first* one in the sorted list
        return currentSeasonIndex > 0;
    })();


    const logoUrl = logoPath ? `https://image.tmdb.org/t/p/w500${logoPath}` : null;


    // -------------------------------------------------------------------
    // 3. PLAYER SCREEN RENDER
    // -------------------------------------------------------------------
    return (
        <div className="fixed top-0 inset-0 bg-black text-white w-full h-full overflow-y-auto animate-fade-in z-[1000]">
            {/* Main container for the split layout */}
            <div className="flex flex-col md:flex-row h-full">

                                )}

                        </div>
                    </div>
                </div>

                {/* Episodes List/Sidebar Section */}
                {hasEpisodes && (
                    <div className="w-full h-full overflow-y-auto p-4 md:w-1/3 md:overflow-y-auto md:h-full md:border-l md:border-white/10">

               </div>

const NARUTO_SHIPPUDEN_START_EPISODES = {
    1: 1, 2: 33, 3: 54, 4: 72, 5: 89, 6: 113, 
    7: 144, 8: 152, 10: 197, 12: 243, 13: 276, 
    14: 296, 15: 321,
};
const NARUTO_SHIPPUDEN_END_EPISODES = {
    1: 32, 2: 53, 3: 71, 4: 88, 5: 112, 6: 143, 
    7: 151, 8: 175, 10: 221, 12: 275, 13: 295,
    14: 320, 15: 348

const EpisodesList = React.memo(({ 
    tvId, 
    seasons, 
    onEpisodeClick, 
    onSeasonSelect, 
    currentPlayingSeasonNumber,
    currentPlayingEpisodeNumber, // Used for highlighting
    selectedSeason, // Passed from parent for sync
    handleSeasonChange, // Passed from parent for control
    isSeasonListOpen,
    setIsSeasonListOpen,
    episodeFilter // NEW: filter string for search
}) => {
    if (!tvId) return null;

    // Local state for episodes and loading moved to parent (DedicatedPlayerScreen) for filtering/header control
    const [episodes, setEpisodes] = useState([]);
    const [loadingEpisodes, setLoadingEpisodes] = useState(false);
    const [error, setError] = useState(null);
    
    const MAX_SEASON_NAME_LENGTH = 20;

    const fetchEpisodes = useCallback(async (seasonNumber) => {
        if (!tvId || !seasonNumber) return;

        setLoadingEpisodes(true);
        setError(null);
        try {
            const res = await fetch(
                `${API_BASE_URL}/tv/${tvId}/season/${seasonNumber}?api_key=${API_KEY}`
            );
            const data = await res.json();

            if (data.success === false) {
                setError("Failed to load episodes for this season.");
                setEpisodes([]);
                const currentSeasonData = seasons.find(s => s.season_number === seasonNumber) || null;
                onSeasonSelect(currentSeasonData);
                return;
            }

            setEpisodes(data.episodes || []);
            // Pass the entire season details object back to the parent
            onSeasonSelect(data); 

        } catch (err) {
            setError("Error fetching episodes.");
        } finally {
            setLoadingEpisodes(false);
        }
    }, [tvId, seasons, onSeasonSelect]);

    // Effect: Fetch episodes whenever selectedSeason changes
    useEffect(() => {
        if (selectedSeason) {
            fetchEpisodes(selectedSeason.season_number);
        }
    }, [selectedSeason, fetchEpisodes]);

    // Effect: Set initial season on mount or when seasons change (Keep this logic simple)
    // NOTE: Initialization of selectedSeason state is now handled in DedicatedPlayerScreen 
    // to centralize control, but we keep the fetch trigger here.
    
    // --- Filtering Logic ---
    const filteredEpisodes = episodes.filter(episode => {
        if (!episodeFilter) return true;
        const searchLower = episodeFilter.toLowerCase();
        
        return (
            (episode.name && episode.name.toLowerCase().includes(searchLower)) ||
            (episode.overview && episode.overview.toLowerCase().includes(searchLower)) ||
            (episode.episode_number.toString().includes(searchLower))
        );
    });

    const formatSeasonName = (name) => {
        if (!name) return '';
        if (name.length > MAX_SEASON_NAME_LENGTH) {
            return name.slice(0, MAX_SEASON_NAME_LENGTH - 3) + '...';
        }
        return name;
    };

    const availableSeasons = seasons.filter(s => s.season_number !== 0);

    // Memoize the episode click handler for better performance
    const handleEpisodeClick = useCallback((episode) => {
        onEpisodeClick(episode);
    }, [onEpisodeClick]);

    return (
        <div>
            {/* The Season Selector dropdown is moved to the parent (DedicatedPlayerScreen) 
                to allow the 'Prev/Next' buttons to wrap around it. */}
            
            {loadingEpisodes && <div className="text-pink-500">Loading episodes...</div>}
            {error && <div className="text-red-500">{error}</div>}

            {!loadingEpisodes && !error && (
                <div className="space-y-3"> {/* Reduced spacing to match image */}
                    {filteredEpisodes.length > 0 ? (
                        filteredEpisodes.map((episode) => {
                            const isCurrentEpisode = 
                                episode.season_number === currentPlayingSeasonNumber && 
                                episode.episode_number === currentPlayingEpisodeNumber;

                            return (
                                <div 
                                    key={episode.id} 
                                    // UPDATED STYLING: Minimal background, border for active episode, border-transparent default
                                    className={`
                                        flex gap-4 p-2 rounded-lg transition cursor-pointer 
                                        ${isCurrentEpisode 
                                            ? 'bg-pink-600/20 border border-pink-600' // Highlight active episode
                                            : 'bg-transparent hover:bg-white/5 border border-transparent' // Default style
                                        }
                                    `}
                                    onClick={() => handleEpisodeClick(episode)} 
                                >
                                    {/* Thumbnail Section */}
                                    <div className="relative flex-shrink-0 w-32 h-20 rounded-lg overflow-hidden bg-black">
                                        <img
                                            src={
                                                episode.still_path
                                                    ? `https://image.tmdb.org/t/p/w300${episode.still_path}`
                                                    : '/No-Poster.png'
                                            }
                                            alt={`S${episode.season_number} E${episode.episode_number}`}
                                            className="w-full h-full object-cover"
                                        />
                                        {/* Play icon REMOVED to match image style */}
                                    </div>

                                    {/* Text Content Section */}
                                    <div className="flex-grow flex flex-col justify-center">
                                        {/* Episode Number and Title */}
                                        <h3 className="font-medium text-sm lg:text-base line-clamp-2">
                                            {episode.episode_number}. {episode.name}
                                        </h3>
                                        
                                        {/* Date Text (mimicking image style) */}
                                        <p className="text-xs text-gray-400 mt-1">
                                            {episode.air_date 
                                                ? new Date(episode.air_date).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' }) 
                                                : 'Date N/A'}
                                        </p>
                                    </div>
                                    {/* Runtime and Overview REMOVED */}
                                </div>
                            );
                        })
                    ) : (
                        <div className="text-gray-400 p-4">
                            {episodeFilter ? `No episodes found matching "${episodeFilter}".` : 'No episodes found for this season.'}
                        </div>
                    )}
                </div>
            )}
        </div>
    );
});

const DedicatedPlayerScreen = ({ movie, isTV }) => { 

    const navigate = useNavigate();
    const [searchParams, setSearchParams] = useSearchParams(); 

    // Guard Clause
    if (!movie) {
        useEffect(() => {
            navigate("/", { replace: true });
        }, [navigate]);
        return null;
    }

    // Initialize state from URL Search Params
    const initialSeason = parseInt(searchParams.get('s')) || 1;
    const initialEpisode = parseInt(searchParams.get('e')) || 1;

    const [seasons, setSeasons] = useState(movie.seasons || []);
    // currentSeasonDetails will hold the full season object (including episodes array) from TMDB API
    const [currentSeasonDetails, setCurrentSeasonDetails] = useState(null); 
    const [playingEpisode, setPlayingEpisode] = useState({
        season_number: initialSeason,
        episode_number: initialEpisode,
    });
    const [logoPath, setLogoPath] = useState(null);
    
    // NEW STATE for Episode List control
    const [selectedSeason, setSelectedSeason] = useState(null); // Used to control fetch in EpisodesList
    const [isSeasonListOpen, setIsSeasonListOpen] = useState(false); 
    const [episodeFilter, setEpisodeFilter] = useState(''); // Search bar filter

    const isAnilist = movie.media_type === 'anilist' || !!movie?.title?.romaji;
    const tmdbId = isAnilist ? null : movie.id; 

    // Helper to update the episode state AND the URL search parameters
    const updatePlayingEpisode = useCallback((season, episode) => {
        setPlayingEpisode({
            season_number: season,
            episode_number: episode,
        });

        // Update URL search parameters
        setSearchParams({ s: season, e: episode }, { replace: true });
    }, [setSearchParams]); 


    // Combined Embed URL Logic - **Uses playingEpisode state**
    const getEmbedUrl = useCallback(() => {
        const links = CUSTOM_EMBED_LINKS[movie.id];
        const episodeNum = playingEpisode?.episode_number || 1; // Uses state
        const seasonNum = playingEpisode?.season_number || 1; // Uses state

        if (links) {
            if (typeof links === 'object' && !Array.isArray(links)) {
                const seasonLinks = links[seasonNum];

                if (seasonLinks && Array.isArray(seasonLinks)) {
                    let linkIndex = episodeNum - 1; 

                    if (movie.id === 31910) {
                        const startEpisode = NARUTO_SHIPPUDEN_START_EPISODES[seasonNum] || 1;
                        // For Naruto, linkIndex is the offset from the starting TMDB episode number
                        linkIndex = episodeNum - startEpisode; 
                    }

                    if (linkIndex >= 0 && linkIndex < seasonLinks.length) {
                        let url = seasonLinks[linkIndex];
                        // Normalize Google Drive link for embedding
                        if (url.includes('drive.google.com') && !url.includes('/preview')) {
                            url = url.replace('/view?usp=drivesdk', '/preview');
                            url = url.replace(/\/file\/d\/([\w-]+)\/view/g, '/file/d/$1/preview');
                        }
                        return url;
                    }
                }
            }

            if (Array.isArray(links)) {
                return links[episodeNum - 1] || links[0];
            }

            if (typeof links === 'string') {
                return links;
            }
        }

        // Fallback to generic embed provider (Vidplus.to)
        if (isAnilist) {
            return `https://player.vidplus.to/embed/anime/${movie.id}`;
        } else if (isTV) {
            return `https://player.vidplus.to/embed/tv/${movie.id}-${seasonNum}-${episodeNum}`; 
        } else {
            return `https://player.vidplus.to/embed/movie/${movie.id}`;
        }
    }, [movie.id, isAnilist, isTV, playingEpisode]);

    const embedUrl = getEmbedUrl();

    // Handle full season details coming back from EpisodesList component
    const handleSeasonSelect = useCallback((seasonDetails) => {
        setCurrentSeasonDetails(seasonDetails);
    }, []);

    // **MEMOIZED** Episode Navigation Logic (Next Episode)
    const handleNextEpisode = useCallback(() => {
        if (!playingEpisode || !isTV || !currentSeasonDetails) return;

        const currentSeasonNum = playingEpisode.season_number;
        const currentEpisodeNum = playingEpisode.episode_number;
        const nextEpisodeNum = currentEpisodeNum + 1;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);

        let maxEpisodeInCurrentSeason;
        if (movie.id === 31910) {
            maxEpisodeInCurrentSeason = NARUTO_SHIPPUDEN_END_EPISODES[currentSeasonNum];
        } else {
            // Use the episode count from the currentSeasonDetails API response
            maxEpisodeInCurrentSeason = currentSeasonDetails.episodes ? currentSeasonDetails.episodes.length : 0;
        }

        if (nextEpisodeNum <= maxEpisodeInCurrentSeason) {
            // Play next episode in the same season
            updatePlayingEpisode(currentSeasonNum, nextEpisodeNum); 
        } else {
            // Check for the next season
            const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
            const nextSeason = allSeasons[currentSeasonIndex + 1];

            if (nextSeason) {
                let nextSeasonStartEpisode = 1; 
                if (movie.id === 31910) {
                    nextSeasonStartEpisode = NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number] || 1;
                }
                updatePlayingEpisode(nextSeason.season_number, nextSeasonStartEpisode); 
            } else {
                alert("Congratulations! You've reached the end of the available episodes.");
            }
        }
    }, [playingEpisode, isTV, currentSeasonDetails, seasons, movie.id, updatePlayingEpisode]);

    // Handle jump to next season (UPDATED to use updatePlayingEpisode)
    const handleNextSeason = useCallback(() => {
        if (!isTV || !playingEpisode) return;

        const currentSeasonNum = playingEpisode.season_number;
        // Filter out Season 0 (Specials) and sort by season number
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
        const nextSeason = allSeasons[currentSeasonIndex + 1];

        if (nextSeason) {
            let nextSeasonStartEpisode = 1; 
            if (movie.id === 31910) {
                nextSeasonStartEpisode = NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number] || 1;
            }

            // Update state and URL to the first episode of the next season
            updatePlayingEpisode(nextSeason.season_number, nextSeasonStartEpisode);
            // Also update local selectedSeason state for EpisodesList synchronization
            setSelectedSeason(nextSeason);
        } else {
            alert("You've reached the final season.");
        }
    }, [isTV, playingEpisode, seasons, movie.id, updatePlayingEpisode]); 

    // Handle jump to previous season (UPDATED to use updatePlayingEpisode)
    const handlePreviousSeason = useCallback(() => {
        if (!isTV || !playingEpisode) return;

        const currentSeasonNum = playingEpisode.season_number;
        // Filter out Season 0 (Specials) and sort by season number
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);
        const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
        const previousSeason = allSeasons[currentSeasonIndex - 1]; // Look one index back

        if (previousSeason) {
            let targetEpisode = 1; 
            if (movie.id === 31910) {
                // For Naruto, navigate to the last episode of the previous season
                targetEpisode = NARUTO_SHIPPUDEN_END_EPISODES[previousSeason.season_number] || 1;
            }
            // For others, we default to episode 1 of the previous season.

            // Update state and URL to the target episode of the previous season
            updatePlayingEpisode(previousSeason.season_number, targetEpisode);
            // Also update local selectedSeason state for EpisodesList synchronization
            setSelectedSeason(previousSeason);
        } else {
            alert("You are already on the first season.");
        }
    }, [isTV, playingEpisode, seasons, movie.id, updatePlayingEpisode]);


    // Update handleEpisodeClick to use the new helper and update URL
    const handleEpisodeClick = useCallback((episode) => {
        updatePlayingEpisode(episode.season_number, episode.episode_number);
    }, [updatePlayingEpisode]);

    // Memoize handleBackToDetails
    const handleBackToDetails = useCallback(() => {
        navigate(-1); 
    }, [navigate]);

    // Initial Season Setter (Moved from EpisodesList)
    useEffect(() => {
        if (seasons && seasons.length > 0) {
            const initialSeasonNumber = playingEpisode.season_number || 
                                        seasons.find(s => s.season_number > 0)?.season_number || 
                                        seasons[0].season_number;

            const initialSeason = seasons.find(s => s.season_number === initialSeasonNumber);

            if (initialSeason && initialSeason.id !== selectedSeason?.id) {
                setSelectedSeason(initialSeason);
            }
        }
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [seasons, playingEpisode.season_number]); // Depend on seasons and playingEpisode.season_number

    // Handle Season Dropdown Change
    const handleSeasonDropdownChange = useCallback((season) => {
        if (season.id !== selectedSeason?.id) {
            setSelectedSeason(season);
            setIsSeasonListOpen(false); 
            // Optional: When season changes, default to episode 1 in the URL/state
            updatePlayingEpisode(season.season_number, 1);
        }
    }, [selectedSeason, updatePlayingEpisode]);

    // Fetch details for seasons/logo if not already available
    useEffect(() => {
        if (isAnilist || !tmdbId) return;

        const fetchExtraDetails = async () => {

                    setSeasons(data.seasons);

                const logos = data.images?.logos || [];
                const englishLogo = logos.find(logo => logo.iso_639_1 === 'en') || logos[0];
                setLogoPath(englishLogo?.file_path || null);

            } catch (err) {
                console.error("Failed to fetch player extra details:", err);
            }
        };
        fetchExtraDetails();
    }, [movie, isAnilist, seasons.length, tmdbId]);


    const title = isAnilist ? movie.title.english || movie.title.romaji : movie.title || movie.name;
    const hasEpisodes = seasons.length > 0 && !isAnilist && isTV; 
    const availableSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);

            // Use episodes array from the full season details

            return true;
        }

        const nextSeason = availableSeasons[currentSeasonIndex + 1];

        // If current episode is the last one, check if there's a next season.
        if (nextSeason) {
            if (movie.id === 31910) {
                // For Naruto, ensure the next season has a defined start episode
                return !!NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number];
            }
            return true;
        }

        return false;
    })();

    // Logic to determine if there is a next season available (UPDATED to use currentSeasonIndex)
    const canSkipToNextSeason = currentSeasonIndex !== -1 && currentSeasonIndex < availableSeasons.length - 1;

    // Logic to determine if there is a previous season available (UPDATED to use currentSeasonIndex)
    const canSkipToPreviousSeason = currentSeasonIndex > 0;

    const logoUrl = logoPath ? `https://image.tmdb.org/t/p/w500${logoPath}` : null;

                            title="Content Player"
                            allowFullScreen
                            className="w-full h-full"
                            key={embedUrl} // Key forces iframe reload on episode change
                        />

                        {/* Back Button (Keep original style as it's outside the sidebar scope) */}
                        <button
                            className="absolute top-4 left-4 bg-black/70 px-4 py-2 rounded-lg hover:bg-pink-600 transition flex items-center gap-1 text-sm z-50"
                            onClick={handleBackToDetails}

                            <div className="flex flex-wrap gap-4 text-xs mt-2 sm:mt-0"> 
                                
                                {/* NEXT EPISODE BUTTON */}

const EpisodesList = React.memo(({ 
    tvId, 
    seasons, 
    onEpisodeClick, 
    onSeasonSelect, 
    currentPlayingSeasonNumber,
    currentPlayingEpisodeNumber, 
    selectedSeason, 
    episodeFilter 
}) => {
    if (!tvId) return null;

    const [episodes, setEpisodes] = useState([]);
    const [loadingEpisodes, setLoadingEpisodes] = useState(false);
    const [error, setError] = useState(null);

    const fetchEpisodes = useCallback(async (seasonNumber) => {
        if (!tvId || !seasonNumber) return;

        setLoadingEpisodes(true);
        setError(null);
        try {
            const res = await fetch(
                `${API_BASE_URL}/tv/${tvId}/season/${seasonNumber}?api_key=${API_KEY}`
            );
            const data = await res.json();

            if (data.success === false) {
                setError("Failed to load episodes for this season.");
                setEpisodes([]);
                const currentSeasonData = seasons.find(s => s.season_number === seasonNumber) || null;
                onSeasonSelect(currentSeasonData);
                return;
            }

            setEpisodes(data.episodes || []);
            onSeasonSelect(data); 

        } catch (err) {
            setError("Error fetching episodes.");
        } finally {
            setLoadingEpisodes(false);
        }
    }, [tvId, seasons, onSeasonSelect]);

    // Effect: Fetch episodes whenever selectedSeason changes
    useEffect(() => {
        if (selectedSeason) {
            fetchEpisodes(selectedSeason.season_number);
        }
    }, [selectedSeason, fetchEpisodes]);

    // --- Filtering Logic ---
    const filteredEpisodes = episodes.filter(episode => {
        if (!episodeFilter) return true;
        const searchLower = episodeFilter.toLowerCase();

        return (
            (episode.name && episode.name.toLowerCase().includes(searchLower)) ||
            (episode.overview && episode.overview.toLowerCase().includes(searchLower)) ||
            (episode.episode_number.toString().includes(searchLower))
        );
    });

    const handleEpisodeClick = useCallback((episode) => {
        onEpisodeClick(episode);
    }, [onEpisodeClick]);

    return (
        <div>
            {loadingEpisodes && <div className="text-pink-500">Loading episodes...</div>}
            {error && <div className="text-red-500">{error}</div>}

            {!loadingEpisodes && !error && (
                <div className="space-y-3"> 
                    {filteredEpisodes.length > 0 ? (
                        filteredEpisodes.map((episode) => {
                            const isCurrentEpisode = 
                                episode.season_number === currentPlayingSeasonNumber && 
                                episode.episode_number === currentPlayingEpisodeNumber;

                            return (
                                <div 
                                    key={episode.id} 
                                    className={`
                                        flex gap-4 p-2 rounded-lg transition cursor-pointer 
                                        ${isCurrentEpisode 
                                            ? 'bg-pink-600/20 border border-pink-600' 
                                            : 'bg-transparent hover:bg-white/5 border border-transparent' 
                                        }
                                    `}
                                    onClick={() => handleEpisodeClick(episode)} 
                                >
                                    {/* Thumbnail Section */}
                                    <div className="relative flex-shrink-0 w-32 h-20 rounded-lg overflow-hidden bg-black">
                                        <img
                                            src={
                                                episode.still_path
                                                    ? `https://image.tmdb.org/t/p/w300${episode.still_path}`
                                                    : '/No-Poster.png'
                                            }
                                            alt={`S${episode.season_number} E${episode.episode_number}`}
                                            className="w-full h-full object-cover"
                                        />
                                    </div>

                                    {/* Text Content Section */}
                                    <div className="flex-grow flex-col justify-center">
                                        {/* Episode Number and Title */}
                                        <h3 className="font-medium text-sm lg:text-base line-clamp-2">
                                            {episode.episode_number}. {episode.name}
                                        </h3>

                                        {/* Date Text */}
                                        <p className="text-xs text-gray-400 mt-1">
                                            {episode.air_date 
                                                ? new Date(episode.air_date).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' }) 
                                                : 'Date N/A'}
                                        </p>
                                    </div>
                                </div>
                            );
                        })
                    ) : (
                        <div className="text-gray-400 p-4">
                            {episodeFilter ? `No episodes found matching "${episodeFilter}".` : 'No episodes found for this season.'}
                        </div>
                    )}
                </div>
            )}
        </div>
    );
});

const DedicatedPlayerScreen = ({ movie, isTV }) => { 

    const navigate = useNavigate();

        useEffect(() => {
            navigate("/", { replace: true });
        }, [navigate]);

    }

    const initialSeason = parseInt(searchParams.get('s')) || 1;
    const initialEpisode = parseInt(searchParams.get('e')) || 1;

    const [seasons, setSeasons] = useState(movie.seasons || []);
    const [currentSeasonDetails, setCurrentSeasonDetails] = useState(null); 
    const [playingEpisode, setPlayingEpisode] = useState({
        season_number: initialSeason,
        episode_number: initialEpisode,
    });
    const [logoPath, setLogoPath] = useState(null);
    const [showIframe, setShowIframe] = useState(false); // 💡 NEW STATE for poster overlay

    const [selectedSeason, setSelectedSeason] = useState(null); 
    const [isSeasonListOpen, setIsSeasonListOpen] = useState(false); 
    const [episodeFilter, setEpisodeFilter] = useState(''); 

    const isAnilist = movie.media_type === 'anilist' || !!movie?.title?.romaji;
    const tmdbId = isAnilist ? null : movie.id; 

    const updatePlayingEpisode = useCallback((season, episode) => {
        setPlayingEpisode({
            season_number: season,
            episode_number: episode,
        });
        setSearchParams({ s: season, e: episode }, { replace: true });
        setShowIframe(false); // Reset to poster when episode changes
    }, [setSearchParams]); 

    // Effect to fetch the logo path
    useEffect(() => {
        const fetchLogo = async () => {
            if (!tmdbId || isAnilist) return;
            try {
                const res = await fetch(`${API_BASE_URL}/tv/${tmdbId}/images?api_key=${API_KEY}`);
                const data = await res.json();
                const logo = data.logos?.find(l => l.iso_639_1 === 'en') || data.logos?.[0];
                if (logo && logo.file_path) {
                    setLogoPath(logo.file_path);
                }
            } catch (error) {
                console.error("Failed to fetch logo:", error);
            }
        };
        fetchLogo();
    }, [tmdbId, isAnilist]);

    const getEmbedUrl = useCallback(() => {
        let links = CUSTOM_EMBED_LINKS[movie.id];
        if (!links) {
            links = CUSTOM_EMBED_LINKS_2[movie.id];
        }

        const episodeNum = playingEpisode?.episode_number || 1; 
        const seasonNum = playingEpisode?.season_number || 1; 

        if (links) {
            if (typeof links === 'object' && !Array.isArray(links)) {
                const seasonLinks = links[seasonNum];

                if (seasonLinks && Array.isArray(seasonLinks)) {
                    let linkIndex = episodeNum - 1; 

                    // 🎯 START: Custom Link Index Logic for 31910 and 46260
                    if (movie.id === 31910 || movie.id === 46260) {
                        let startEpisode = 1;

                        if (movie.id === 31910) {
                            startEpisode = NARUTO_SHIPPUDEN_START_EPISODES[seasonNum] || 1;
                        } else if (movie.id === 46260) {
                            startEpisode = TV_46260_START_EPISODES[seasonNum] || 1;
                        }

                        linkIndex = episodeNum - startEpisode; 
                    }
                    // 🎯 END: Custom Link Index Logic

                    if (linkIndex >= 0 && linkIndex < seasonLinks.length) {
                        let url = seasonLinks[linkIndex];
                        if (url.includes('drive.google.com') && !url.includes('/preview')) {
                            url = url.replace('/view?usp=drivesdk', '/preview');

                        }
                        return url;
                    }
                }
            }

            if (Array.isArray(links)) {
                return links[episodeNum - 1] || links[0];
            }
            if (typeof links === 'string') {
                return links;
            }
        }

        if (isAnilist) {
            return `https://player.vidplus.to/embed/anime/${movie.id}`;
        } else if (isTV) {
            return `https://player.vidplus.to/embed/tv/${movie.id}-${seasonNum}-${episodeNum}`;

            return `https://player.vidplus.to/embed/movie/${movie.id}`;

        setCurrentSeasonDetails(seasonDetails);

    // 💡 NEW HELPER: Determine the minimum episode number for the current season

        if (movie.id === 31910) {
            return NARUTO_SHIPPUDEN_START_EPISODES[seasonNum] || 1;
        } else if (movie.id === 46260) {
            return TV_46260_START_EPISODES[seasonNum] || 1;
        }
        return 1;
    }, [movie.id]);

    // 💡 NEW HELPER: Determine the maximum episode number for the current season
    const getCurrentSeasonMaxEpisode = useCallback((seasonNum) => {
        if (movie.id === 31910) {
            return NARUTO_SHIPPUDEN_END_EPISODES[seasonNum];
        } else if (movie.id === 46260) {
            return TV_46260_END_EPISODES[seasonNum];
        } else {
            // Fallback to TMDB data length if available and not using custom maps
            return currentSeasonDetails?.episodes?.length || 0;
        }
    }, [movie.id, currentSeasonDetails]);


    // 🎯 START: Next Episode Logic (Enhanced with new helpers)

        if (!playingEpisode || !isTV) return;

        const currentSeasonNum = playingEpisode.season_number;
        const currentEpisodeNum = playingEpisode.episode_number;
        const nextEpisodeNum = currentEpisodeNum + 1;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);

        const maxEpisodeInCurrentSeason = getCurrentSeasonMaxEpisode(currentSeasonNum);

        if (maxEpisodeInCurrentSeason && nextEpisodeNum <= maxEpisodeInCurrentSeason) {
            updatePlayingEpisode(currentSeasonNum, nextEpisodeNum);

            const nextSeason = allSeasons[currentSeasonIndex + 1];

                const targetEpisode = prevSeasonMaxEpisode || 1;

                alert("You are already on the first episode of the first season.");

const EpisodesList = React.memo(({ 
    tvId, 
    seasons, 
    onEpisodeClick, 
    onSeasonSelect, 
    currentPlayingSeasonNumber,
    currentPlayingEpisodeNumber, 
    selectedSeason, 
    episodeFilter 
}) => {
    if (!tvId) return null;

    const [episodes, setEpisodes] = useState([]);
    const [loadingEpisodes, setLoadingEpisodes] = useState(false);
    const [error, setError] = useState(null);

    const fetchEpisodes = useCallback(async (seasonNumber) => {
        if (!tvId || !seasonNumber) return;

        setLoadingEpisodes(true);
        setError(null);
        try {
            const res = await fetch(
                `${API_BASE_URL}/tv/${tvId}/season/${seasonNumber}?api_key=${API_KEY}`
            );
            const data = await res.json();

            if (data.success === false) {
                setError("Failed to load episodes for this season.");
                setEpisodes([]);
                const currentSeasonData = seasons.find(s => s.season_number === seasonNumber) || null;
                onSeasonSelect(currentSeasonData);
                return;
            }

            setEpisodes(data.episodes || []);
            onSeasonSelect(data); 

        } catch (err) {
            setError("Error fetching episodes.");
        } finally {
            setLoadingEpisodes(false);
        }
    }, [tvId, seasons, onSeasonSelect]);

    // Effect: Fetch episodes whenever selectedSeason changes
    useEffect(() => {
        if (selectedSeason) {
            fetchEpisodes(selectedSeason.season_number);
        }
    }, [selectedSeason, fetchEpisodes]);

    // --- Filtering Logic ---
    const filteredEpisodes = episodes.filter(episode => {
        if (!episodeFilter) return true;
        const searchLower = episodeFilter.toLowerCase();

        return (
            (episode.name && episode.name.toLowerCase().includes(searchLower)) ||
            (episode.overview && episode.overview.toLowerCase().includes(searchLower)) ||
            (episode.episode_number.toString().includes(searchLower))
        );
    });

    const handleEpisodeClick = useCallback((episode) => {
        onEpisodeClick(episode);
    }, [onEpisodeClick]);

    return (
        <div>
            {loadingEpisodes && <div className="text-pink-500">Loading episodes...</div>}
            {error && <div className="text-red-500">{error}</div>}

            {!loadingEpisodes && !error && (
                <div className="space-y-3"> 
                    {filteredEpisodes.length > 0 ? (
                        filteredEpisodes.map((episode) => {
                            const isCurrentEpisode = 
                                episode.season_number === currentPlayingSeasonNumber && 
                                episode.episode_number === currentPlayingEpisodeNumber;

                            return (
                                <div 
                                    key={episode.id} 
                                    className={`
                                        flex gap-4 p-2 rounded-lg transition cursor-pointer 
                                        ${isCurrentEpisode 
                                            ? 'bg-pink-600/20 border border-pink-600' 
                                            : 'bg-transparent hover:bg-white/5 border border-transparent' 
                                        }
                                    `}
                                    onClick={() => handleEpisodeClick(episode)} 
                                >
                                    {/* Thumbnail Section */}
                                    <div className="relative flex-shrink-0 w-32 h-20 rounded-lg overflow-hidden bg-black">
                                        <img
                                            src={
                                                episode.still_path
                                                    ? `https://image.tmdb.org/t/p/w300${episode.still_path}`
                                                    : '/No-Poster.png'
                                            }
                                            alt={`S${episode.season_number} E${episode.episode_number}`}
                                            className="w-full h-full object-cover"
                                        />
                                    </div>

                                    {/* Text Content Section */}
                                    <div className="flex-grow flex-col justify-center">
                                        {/* Episode Number and Title */}
                                        <h3 className="font-medium text-sm lg:text-base line-clamp-2">
                                            {episode.episode_number}. {episode.name}
                                        </h3>

                                        {/* Date Text */}
                                        <p className="text-xs text-gray-400 mt-1">
                                            {episode.air_date 
                                                ? new Date(episode.air_date).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' }) 
                                                : 'Date N/A'}
                                        </p>
                                    </div>
                                </div>
                            );
                        })
                    ) : (
                        <div className="text-gray-400 p-4">
                            {episodeFilter ? `No episodes found matching "${episodeFilter}".` : 'No episodes found for this season.'}
                        </div>
                    )}
                </div>
            )}
        </div>
    );
});

const DedicatedPlayerScreen = ({ movie, isTV }) => {

        return null;
    }

    const initialSeason = parseInt(searchParams.get('s')) || 1;
    const initialEpisode = parseInt(searchParams.get('e')) || 1;

    const [seasons, setSeasons] = useState(movie.seasons || []);
    const [currentSeasonDetails, setCurrentSeasonDetails] = useState(null); 
    const [playingEpisode, setPlayingEpisode] = useState({
        season_number: initialSeason,
        episode_number: initialEpisode,
    });

    const tmdbId = isAnilist ? null : movie.id;

        setPlayingEpisode({
            season_number: season,
            episode_number: episode,
        });
        setSearchParams({ s: season, e: episode }, { replace: true });
        setShowIframe(false); // Reset to poster when episode changes
    }, [setSearchParams]); 

    // Effect to fetch the logo path
    useEffect(() => {
        const fetchLogo = async () => {
            if (!tmdbId || isAnilist) return;
            try {
                const res = await fetch(`${API_BASE_URL}/tv/${tmdbId}/images?api_key=${API_KEY}`);
                const data = await res.json();
                const logo = data.logos?.find(l => l.iso_639_1 === 'en') || data.logos?.[0];
                if (logo && logo.file_path) {
                    setLogoPath(logo.file_path);
                }
            } catch (error) {
                console.error("Failed to fetch logo:", error);
            }
        };
        fetchLogo();
    }, [tmdbId, isAnilist]);

    const getEmbedUrl = useCallback(() => {
        let links = CUSTOM_EMBED_LINKS[movie.id];
        if (!links) {
            links = CUSTOM_EMBED_LINKS_2[movie.id];
        }

        const episodeNum = playingEpisode?.episode_number || 1; 
        const seasonNum = playingEpisode?.season_number || 1; 

        if (links) {
            if (typeof links === 'object' && !Array.isArray(links)) {
                const seasonLinks = links[seasonNum];

                if (seasonLinks && Array.isArray(seasonLinks)) {
                    let linkIndex = episodeNum - 1; 

                    // 🎯 START: Custom Link Index Logic for 31910 and 46260
                    if (movie.id === 31910 || movie.id === 46260) {
                        let startEpisode = 1;

                        if (movie.id === 31910) {

                        } else if (movie.id === 46260) {
                            startEpisode = TV_46260_START_EPISODES[seasonNum] || 1;

                        if (url.includes('drive.google.com') && !url.includes('/preview')) {

                        }
                        return url;
                    }
                }
            }

            if (Array.isArray(links)) {
                return links[episodeNum - 1] || links[0];
            }
            if (typeof links === 'string') {
                return links;
            }
        }

        if (isAnilist) {
            return `https://player.vidplus.to/embed/anime/${movie.id}`;
        } else if (isTV) {
            return `https://player.vidplus.to/embed/tv/${movie.id}-${seasonNum}-${episodeNum}`;

        const currentEpisodeNum = playingEpisode.episode_number;
        const nextEpisodeNum = currentEpisodeNum + 1;
        const allSeasons = seasons.filter(s => s.season_number > 0).sort((a, b) => a.season_number - b.season_number);

        let maxEpisodeInCurrentSeason;
        if (movie.id === 31910) {
            maxEpisodeInCurrentSeason = NARUTO_SHIPPUDEN_END_EPISODES[currentSeasonNum];
        } else if (movie.id === 46260) {
            maxEpisodeInCurrentSeason = TV_46260_END_EPISODES[currentSeasonNum];
        } else {
            maxEpisodeInCurrentSeason = currentSeasonDetails.episodes ? currentSeasonDetails.episodes.length : 0;
        }
        // 🎯 END: Custom Max Episode Logic

        if (nextEpisodeNum <= maxEpisodeInCurrentSeason) {
            updatePlayingEpisode(currentSeasonNum, nextEpisodeNum); 
        } else {
            const currentSeasonIndex = allSeasons.findIndex(s => s.season_number === currentSeasonNum);
            const nextSeason = allSeasons[currentSeasonIndex + 1];

            if (nextSeason) {
                let nextSeasonStartEpisode = 1;

                if (movie.id === 31910) {
                    nextSeasonStartEpisode = NARUTO_SHIPPUDEN_START_EPISODES[nextSeason.season_number] || 1;
                } else if (movie.id === 46260) {

        }

animate-fade-in"

                                    disabled={!canSkipToPreviousSeason}
                                    className={`
                                        flex items-center text-sm font-semibold p-2 transition duration-200 
                                        ${canSkipToPreviousSeason ? 'text-white hover:text-pink-500' : 'text-gray-600 cursor-not-allowed'}
                                    `}
                                >
                                    <ArrowLeft size={18} className="mr-1" /> Prev
                                </button>

                                {/* Season Dropdown Button */}
                                <div className="relative">
                                    <button
                                        className="flex items-center text-base font-bold transition"
                                        onClick={() => setIsSeasonListOpen(prev => !prev)}
                                    >
                                        <span className="text-lg mr-1 max-w-[140px] whitespace-nowrap overflow-hidden text-ellipsis block">
                                            {selectedSeason ? selectedSeason.name : 'Select Season'}
                                        </span>
                                        <ChevronDown className={`w-4 h-4 transition-transform ${isSeasonListOpen ? 'rotate-180' : 'rotate-0'}`} />
                                    </button>

                                    {/* Dropdown Menu */}
                                    {isSeasonListOpen && (
                                        <div className="absolute right-0 mt-1 w-60 max-h-60 overflow-y-auto bg-gray-900 border border-pink-600 rounded-lg shadow-xl z-20">
                                            {availableSeasons.map((season) => (
                                                <div
                                                    key={season.id}
                                                    className={`py-2 px-4 cursor-pointer text-sm transition-colors ${
                                                        selectedSeason && selectedSeason.id === season.id
                                                            ? 'bg-pink-600 font-semibold' 
                                                            : 'hover:bg-white/10' 
                                                    }`}
                                                    onClick={() => handleSeasonDropdownChange(season)}
                                                >
                                                    {season.name}
                                                </div>
                                            ))}
                                        </div>
                                    )}
                                </div>

                                {/* Next Button */}
                                <button
                                    onClick={handleNextSeason}
                                    disabled={!canSkipToNextSeason}
                                    className={`
                                        flex items-center text-sm font-semibold p-2 transition duration-200 
                                        ${canSkipToNextSeason ? 'text-white hover:text-pink-500' : 'text-gray-600 cursor-not-allowed'}
                                    `}
                                >
                                    Next <ArrowLeft size={18} className="ml-1 rotate-180" />
                                </button>

                            </div>

                            {/* Search Bar */}
                            <div className="relative px-2">
                                <Search className="absolute left-7 top-1/2 -translate-y-1/2 w-4 h-4 text-gray-400" />
                                <input
                                    type="text"
                                    placeholder="Search episodes..."
                                    value={episodeFilter}
                                    onChange={(e) => setEpisodeFilter(e.target.value)}
                                    className="w-full bg-white/5 text-white py-2 pl-10 pr-4 rounded-full placeholder-gray-400 focus:outline-none focus:ring-1 focus:ring-pink-600 border border-transparent"
                                />
                            </div>
                        </div>


                        {/* --- EPISODES LIST --- */} 
                        <div className="p-4 pt-2">
                            <EpisodesList 
                                tvId={movie.id} 
                                seasons={seasons} 
                                onEpisodeClick={handleEpisodeClick}
                                onSeasonSelect={handleSeasonSelect}
                                currentPlayingSeasonNumber={playingEpisode.season_number} 
                                currentPlayingEpisodeNumber={playingEpisode.episode_number}
                                selectedSeason={selectedSeason}
                                episodeFilter={episodeFilter}
                            />
                        </div>
                    </div>
                )}
               </div>
      </div>
    );
};

export default DedicatedPlayerScreen;
